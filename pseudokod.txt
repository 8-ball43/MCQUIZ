1.Sprawdzanie czy użytkownik jest zalogowany

POCZĄTEK
    // Pobranie danych z ciasteczek
    token_user ← POBIERZ Z COOKIE "token"
    id_user ← POBIERZ Z COOKIE "id"

    // Sprawdzenie, czy token w ogóle istnieje
    JEŻELI token_user JEST PUSTY TO:
        PRZEKIERUJ DO "login.php"
        ZAKOŃCZ

    // Weryfikacja z bazą danych
    dane_z_bazy ← WYKONAJ SQL "SELECT token FROM uzytkownicy WHERE id = id_user"
    
    // Sprawdzenie poprawności tokenu
    JEŻELI token_user RÓŻNI SIĘ OD dane_z_bazy.token TO:
        PRZEKIERUJ DO "login.php"
        ZAKOŃCZ
    W PRZECIWNYM RAZIE:
        KONTYNUUJ (użytkownik zalogowany)
KONIEC
